### BookingForm.vue 상세

```vue
<template>
  <div class="booking-form">
    <h3>{{ $t('booking.title') }}</h3>

    <!-- 날짜 선택 -->
    <DatePicker
      v-model="form.date"
      :available-dates="availableDates"
      @change="onDateChange"
    />

    <!-- 인원 선택 -->
    <div class="persons">
      <PersonCounter
        v-model="form.adult_count"
        :label="$t('booking.adults')"
        :min="1"
        :max="maxPersons - form.child_count"
      />
      <PersonCounter
        v-model="form.child_count"
        :label="$t('booking.children')"
        :min="0"
        :max="maxPersons - form.adult_count"
      />
    </div>

    <!-- 옵션 선택 -->
    <div v-if="options.length" class="options">
      <h4>{{ $t('booking.options') }}</h4>
      <label v-for="option in options" :key="option.id">
        <input
          type="checkbox"
          v-model="form.options"
          :value="option.id"
        />
        {{ option.label }} (+{{ formatPrice(option.amount) }})
      </label>
    </div>

    <!-- 가격 요약 -->
    <div class="price-summary">
      <div>{{ $t('booking.subtotal') }}: {{ formatPrice(subtotal) }}</div>
      <div>{{ $t('booking.options_total') }}: {{ formatPrice(optionsTotal) }}</div>
      <div class="total">{{ $t('booking.total') }}: {{ formatPrice(total) }}</div>
    </div>

    <!-- 예약 버튼 -->
    <button
      @click="submitBooking"
      :disabled="!canBook || isSubmitting"
      class="btn-primary"
    >
      {{ isSubmitting ? $t('booking.processing') : $t('booking.submit') }}
    </button>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'
import { useBookingStore } from '@/stores/booking'

const props = defineProps({
  product: Object,
  prices: Array,
  options: Array,
  schedules: Array,
})

const bookingStore = useBookingStore()

const form = ref({
  date: null,
  adult_count: 1,
  child_count: 0,
  options: [],
})

const availableDates = computed(() => {
  return props.schedules
    .filter(s => s.is_active && s.available_count > 0)
    .map(s => s.date)
})

const subtotal = computed(() => {
  const adultPrice = props.prices.find(p => p.type === 'adult')?.amount || 0
  const childPrice = props.prices.find(p => p.type === 'child')?.amount || 0
  return (form.value.adult_count * adultPrice) + (form.value.child_count * childPrice)
})

const submitBooking = async () => {
  await bookingStore.createBooking({
    product_id: props.product.id,
    ...form.value,
  })
}
</script>
```

public function handle($request, Closure $next)
{
    $locale = $request->route('locale');

    if (in_array($locale, ['ko', 'en', 'zh', 'ja'])) {
        app()->setLocale($locale);
    }

    return $next($request);
